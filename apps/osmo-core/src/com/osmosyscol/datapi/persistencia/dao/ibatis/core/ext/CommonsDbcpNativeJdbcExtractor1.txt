package com.osmosyscol.datapi.persistencia.dao.ibatis.core.ext;

import java.lang.reflect.Method;
import java.sql.Connection;

/**
 * Spring modified implementation of the NativeJdbcExtractor interface for Commonsd Dbcp.
 * Returns the underlying Connection to application code instead
 * of wrapper implementation; 
 * @see weblogic.jdbc.extensions.WLConnection#getVendorConnection
 * NOTE: verify that the parameter accessToUnderlyingConnectionAllowed="true" is active 
 */
public class CommonsDbcpNativeJdbcExtractor1 implements INativeJdbcExtractor {
    //private static final String JDBC_EXTENSION_NAME = "org.apache.commons.dbcp.PoolingDataSource$PoolGuardConnectionWrapper";
	private static final String JDBC_EXTENSION_NAME = "org.apache.commons.dbcp.DelegatingConnection";
    
    private final Class<?> jdbcExtensionClass;

    private final Method getVendorConnectionMethod;
    
    /**
     * This constructor retrieves the WebLogic JDBC extension interface,
     * so we can get the underlying vendor connection using reflection.
     */
    public CommonsDbcpNativeJdbcExtractor1() {
        try {
            this.jdbcExtensionClass = getClass().getClassLoader().loadClass(JDBC_EXTENSION_NAME);
            this.getVendorConnectionMethod = this.jdbcExtensionClass.getMethod("getInnermostDelegate", (Class[]) null);
        }
        catch (Exception ex) {
            throw new RuntimeException(
                    "Couldn't initialize CommonsDbcpNativeJdbcExtractor because API classes are not available", ex);
        }
    }

    /* (non-Javadoc)
	 * @see com.osmosyscol.datapi.persistencia.dao.ibatis.core.ext.INativeJdbcExtractor#doGetNativeConnection(java.sql.Connection)
	 */
    @Override
	public Connection doGetNativeConnection(Connection con) throws Exception {
        if (this.jdbcExtensionClass.isAssignableFrom(con.getClass())) {
            try {
            	
            	Connection nativeCon = (Connection) this.getVendorConnectionMethod.invoke(con, (Object[]) null);
                
            	// For some reason, the innermost delegate can be null: not for a
     			// Statement's Connection but for the Connection handle returned by the pool.
     			// We'll fall back to the MetaData's Connection in this case, which is
     			// a native unwrapped Connection with Commons DBCP 1.1 and 1.2.
            	
            	return nativeCon;
            	
                //return (nativeCon != null ? nativeCon : con.getMetaData().getConnection());                
                
                
            }
            catch (Exception ex) {
                throw new RuntimeException("Could not invoke CommonsDbcp getInnermostDelegate method", ex);
            }
        }
        return con;
    }

}
